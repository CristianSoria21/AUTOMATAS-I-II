program uno;
	funtion mayor (char a, char b): boolean 
		
		if (a>=b) then
			return true;
		else
			return false;
		endif
	endf
	
	int r,p;
	r =0;
	p =10;

	repeat
		r=r+2;
	until mayor (r,p);
endp
